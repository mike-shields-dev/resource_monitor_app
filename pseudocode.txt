    // Obtain the hostname from the System

END FUNCTION

FUNCTION get_OS()

END FUNCTION

FUNCTION get_uptime()
    // Obtain the up-time from the System using shell commands
    RETURN uptime in the format "HH:MM" 
END FUNCTION

FUNCTION get_RAM()
    // Obtain the total RAM in appropriate SI units from the System using shell commands
    RETURN total_ram
END FUNCTION

FUNCTION get_free_RAM()
    // Obtain the free RAM in appropriate SI units from the System using shell commands
    RETURN free_RAM
END FUNCTION

FUNCTION get_disk_size()
    // Obtain the primary disk's total capacity in appropriate SI units
    using shell commands
    RETURN disk_size
END FUNCTION

FUNCTION get_disk_space()
    // Obtain the primary disk's free space in appropriate SI units

    RETURN disk_space
END FUNCTION

FUNCTION get_IP_address()

    // using shell commands
    RETURN ip_address

    // Obtain the IP addressing mode of the primary network interface
    // using shell commands
    RETURN ip_mode
